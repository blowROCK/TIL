 var unit = ['', '십', '백','천'];
 var money = ['', '만','억', '경'];

function digit(str){
    var temp = str.split("").reverse().join("");
    var returnStr = [];
    for(var j=0; temp.length>0;j++){
        var forStr = '';
        var a = temp.substr(0,4);
        var test = temp.substr(4, temp.length);
        temp = test;
        for(var i=0; i<a.length; i++){
            if(a[i]!= 0){
                forStr += unit[i]+a[i];
            }
        }
        forStr = forStr.split("").reverse().join("");
        
        if(forStr!= ''){
            returnStr.push(forStr + money[j]);
        }
    }
    return returnStr.reverse().join("")+'원';
}
